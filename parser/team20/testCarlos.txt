CREATE DATABASE IF NOT EXISTS BASE1 MODE 2;
USE BASE1;
CREATE TABLE Usuarios (
    id INTEGER ,
    us1 INTEGER NULL DEFAULT "carlos",
    PRIMARY KEY (id),
    CHECK ( us1>10 )
);
CREATE TYPE PAISES AS ENUM ("Guatemala",'Nicaragua');
UPDATE Usuarios
SET
edad = 1,
ID = 1*2+2
WHERE ID>1;

CREATE TABLE Terreno(
    m VARCHAR(10)
)INHERITS(Usuarios);
